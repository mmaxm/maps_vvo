    Вообще я вижу несколько решений данной задачи.Напишу простейший способ по-моему мнению.
    Для этого давай определимся с тем, что важно в этой задаче:
    а) Касательная к окружности перпендикулярна к радиусу, проведенному в точку касания.
    б) Посмотрим, каким образом можно описать окружность. В данной задаче я использую параметрическое уравнение окружности.
    в) Вспомним уравнение прямой, проведенной через две точки на плоскости.
    Освежив свои знания по данным пунктам и посмотрев на рис. в приложении (я сильно не вырисовывал, но думаю понятно) задача очень легко решается так:
    1. Пусть т.А и т.В точки касания большей и меньшей окружностей к касательной.
    2. Отрезок AD || BC и перпендикулярен АВ, также ВС перпендикулярен АВ. AD=R1 и BC=r2 - радиусы окружностей.
    3. CE - перпендикуляр, опущен на отрезок AD, соответственно CE || AB
    4. alfa - угол наклона прямой DA по отношению к оси X.
    Теперь так: Наша задача найти угол alfa, который через параметрическое уравнение окружности определит нам координату точки A на большей окружности, и координату точки B на меньшей окружности. Через найденные две точки проведем прямую - она же общая касательная обеих окружностей. Итого cos(alfa) = ED/DC (отношение прилегающего катета ED к гипотенузе DC), где ED = R1-r2, DC=R1. Угол найден.

using System.Collections.Generic;
using System.Linq;
using System.Text;
 
namespace TEST_CIRCLE
{
    class Program
    {
        static void Main(string[] args)
        {
            Console.WriteLine("Line coefficients culculation.");
            //Вводим радиусы
            Console.WriteLine("Input R1 => ");
            double R1 = double.Parse(Console.ReadLine());
            Console.WriteLine("Input r2 => ");
            double r2 = double.Parse(Console.ReadLine());
 
            //Решение выполняеться только при R1>=r2
            if (R1 >= r2)
            {
                //Координаты центра большей окружности
                double X0_center_R1 = 0;
                double Y0_center_R1 = 0;
 
                //Координаты центра меньшей окружности
                double X0_center_r2 = R1;
                double Y0_center_r2 = 0;
 
                //Косинус угла
                double angel = (R1 - r2) / R1;
                //Определяем координаты точки касания окружности с касательной для большей окружности (через параметрическое уравнение окружности)
                double X1_R1 = X0_center_R1 + R1 * angel;
                double Y1_R1 = Y0_center_R1 + R1 * Math.Sqrt(1 - Math.Pow(angel, 2));
                //Определяем координаты точки касания окружности с касательной для меньшей окружности(через параметрическое уравнение окружности)
                double X1_r2 = X0_center_r2 + r2 * angel;
                double Y1_r2 = Y0_center_r2 + r2 * Math.Sqrt(1 - Math.Pow(angel, 2));
 
                //Уравнение прямой через две точки (X1_R1, Y1_R1) и (X1_r2, Y1_r2)
                // (X-X1_R1)/X1_r2-X1_R1 = (Y-Y1_R1)/(Y1_r2-Y1_R1)
                //После преобразования: (Y1_R1 - Y1_r2)*X + (X1_r2-X1_R1)*Y + X1_R1*Y1_r2 - X1_r2*Y1_R1 = 0
                //Отсюда коеффициентами будут:
                double A = Math.Round(Y1_R1 - Y1_r2, 2);
                double B = Math.Round(X1_r2 - X1_R1, 2);
                double C = Math.Round(X1_R1 * Y1_r2 - X1_r2 * Y1_R1, 2);
                //Выводим на консоль
                Console.WriteLine(string.Format("Line equation: A*x + B*y + C = 0, where A={0}, B={1}, C={2}", A, B, C));
                Console.ReadKey();
            }
            else
            {
                Console.WriteLine("R1<r2!!!! - Incorrect input.");
                Console.ReadKey();            
            }
            
        }
 
    }
}

        public void CreateGraph(int grids, int radius1, int radius2)
        {
            Graphics graph = pnl_GRAPH.CreateGraphics();
            //Это карандашы от тонкого до толстого для сетки и фигур
            Pen bold_pen = new Pen(Brushes.Black, 3);
            Pen middle_pen = new Pen(Brushes.Black, 2);
            Pen think_pen = new Pen(Brushes.Black, 1);
 
            //Масштаб
            int scale = pnl_GRAPH.Height / grids;
 
            //начало координат
            Point X0Y0 = new Point(pnl_GRAPH.Width / 2, pnl_GRAPH.Height / 2);
 
            //Строим ось Х
            graph.DrawLine(middle_pen, new Point(0, pnl_GRAPH.Height / 2), new Point(pnl_GRAPH.Width, pnl_GRAPH.Height / 2));
            //Строим ось Y
            graph.DrawLine(middle_pen, new Point(pnl_GRAPH.Width / 2, 0), new Point(pnl_GRAPH.Width / 2, pnl_GRAPH.Height));
 
            //Строим координатную сетку вдоль Х
            for (int i = 0; i <= pnl_GRAPH.Height; i++)
            {
                graph.DrawLine(think_pen, new Point(0, i * scale), new Point(pnl_GRAPH.Width, i * scale));
            }
            //Строим координатную сетку вдоль Y
            for (int i = 0; i <= pnl_GRAPH.Width; i++)
            {
                graph.DrawLine(think_pen, new Point(i * scale, 0), new Point(i * scale, pnl_GRAPH.Height));
            }
            //Масштабируем радиусы
            radius1 = radius1 * scale;
            radius2 = radius2 * scale;
            //Рисуем наши окружности
            graph.DrawEllipse(bold_pen, new Rectangle(new Point(X0Y0.X - radius1, X0Y0.Y - radius1), new Size(2 * radius1, 2 * radius1)));
            graph.DrawEllipse(bold_pen, new Rectangle(new Point(X0Y0.X + radius1 - radius2, X0Y0.Y - radius2), new Size(2 * radius2, 2 * radius2)));
 
            //Угол
            double angel = 0;
            if (radius2 >= 2* radius1)
            {
                MessageBox.Show("Касательная вырождаеться в точку!");
                return;
            }
            else if (radius1 > radius2)
            {
                angel = (((double)radius1 - (double)radius2) / (double)radius1);
            }
            else 
            {
                angel = - (((double)radius2 - (double)radius1) / (double)radius2);            
            }
 
            //Это будет точка касания первой окружности
            Point N = new Point();            
            N.X = X0Y0.X - Convert.ToInt16(radius1 * -(angel));
            N.Y = X0Y0.Y - Convert.ToInt16(radius1 * Math.Sqrt(1 - Math.Pow(-angel, 2)));
            //Рисуем линию(от цента до точки касания на окружности)
            graph.DrawLine(think_pen, X0Y0, N);
 
            //Это будет точка касания второй окружности
            Point M = new Point();
            M.X = X0Y0.X + radius1 - Convert.ToInt16(radius2 * -(angel));
            M.Y = X0Y0.Y - Convert.ToInt16(radius2 * Math.Sqrt(1 - Math.Pow(-angel, 2)));
            //Рисуем линию(от цента до точки касания на окружности)
            graph.DrawLine(think_pen, new Point(X0Y0.X + radius1, X0Y0.Y), M);
 
            //Считаем коеффициенты
            double A = ((double)N.Y - (double)M.Y);
            double B = ((double)M.X - (double)N.X);
            double C = ((double)N.X * (double)M.Y - (double)M.X * (double)N.Y);
 
            //Отображаем коэффициенты
            label8.Text = string.Format("A={0}, B={1}, C={2}", A, B, C);
 
            //f(x) при х=0
            int y1 = Convert.ToInt16(-C / B);
            //f(x) при х = width of panel
            int y2 = Convert.ToInt16((-A * pnl_GRAPH.Width - C) / B);
 
            //рисуем касательную
            graph.DrawLine(bold_pen, new Point(0, y1), new Point(pnl_GRAPH.Width, y2));
 
        }

    В случае когда r2 = 2R1 окружности будут иметь одну общую точку, через которую можно провести касательную. Она будет || оси Y. А в случае r2>2R1 общей касательной вообще не будет, так как окружность R1 будет внутри r2.
    Origamy это конечно нужно учесть в задаче.
